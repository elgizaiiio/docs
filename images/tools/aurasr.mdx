## How to Use

### Basic Super-Resolution

```python
import requests

response = requests.post(
    "https://api.megsyai.com/v1/images/super-resolution",
    headers={"Authorization": "Bearer YOUR_API_KEY"},
    files={"image": open("image.png", "rb")},
    data={
        "tool": "aurasr",
        "scale": 4
    }
)

result_url = response.json()["data"]["url"]
```

## API Reference

### Complete Example

```python
import requests

def super_resolve_image(image_path, scale=4, **options):
    with open(image_path, 'rb') as f:
        response = requests.post(
            "https://api.megsyai.com/v1/images/super-resolution",
            headers={"Authorization": "Bearer YOUR_API_KEY"},
            files={"image": f},
            data={
                "tool": "aurasr",
                "scale": scale,
                "mode": options.get("mode", "balanced"),
                "detail_level": options.get("detail_level", "high"),
                "texture_synthesis": options.get("texture_synthesis", True),
                "texture_strength": options.get("texture_strength", 0.7),
                "edge_enhancement": options.get("edge_enhancement", 0.7),
                "artifact_removal": options.get("artifact_removal", True),
                "color_enhancement": options.get("color_enhancement", 0.5)
            }
        )
    return response.json()

# Usage
result = super_resolve_image(
    "photo.jpg",
    scale=4,
    mode="detail",
    detail_level="maximum",
    texture_synthesis=True
)

print(f"Enhanced image: {result['data']['url']}")
```
