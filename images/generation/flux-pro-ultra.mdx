## API Reference

### Complete Example

```python
import requests

def generate_ultra_image(prompt, **options):
    payload = {
        "model": "flux-pro-ultra",
        "prompt": prompt,
        "negative_prompt": options.get("negative_prompt", ""),
        "size": options.get("size", "2048x2048"),
        "quality": options.get("quality", "ultra"),
        "detail_level": options.get("detail_level", "maximum"),
        "refinement_steps": options.get("refinement_steps", 100),
        "guidance_scale": options.get("guidance_scale", 8.0),
        "color_profile": options.get("color_profile", "Adobe RGB"),
        "format": options.get("format", "PNG"),
        "upscale": options.get("upscale", False),
        "auto_enhance": options.get("auto_enhance", True)
    }
    
    response = requests.post(
        "https://api.megsyai.com/v1/images/generate",
        headers={"Authorization": "Bearer YOUR_API_KEY"},
        json=payload
    )
    
    return response.json()

# Usage for commercial photography
result = generate_ultra_image(
    prompt="Professional product photography...",
    size="4096x4096",
    quality="ultra+",
    detail_level="ultra",
    refinement_steps=150,
    color_profile="Adobe RGB",
    upscale=True
)
```
